311 TChain* LAPPDTree::AddFiles(const char* inputdir, bool addsubfolders){
312  // Add files by pattern, directory. Optionally check subdirectories, but only 1 level deep.
313  if(strcmp(inputdir,"")==0){ cerr<<"no wcsim_lappd file to load"<<endl; return nullptr; }
314  if(verbose){
315  cout<<"getting trees from file "<<inputdir;
316  addsubfolders ? cout<<", including first level subdirectories"<<endl : cout<<endl;
317  }
318 
319  // first check if we've been passed a file, directory or a pattern
320  bool isdir;
321  struct stat s;
322  if(stat(inputdir,&s)==0){
323  if(s.st_mode & S_IFDIR){ // mask to extract if it's a directory
324  isdir=true; //it's a directory
325  } else if(s.st_mode & S_IFREG){ // mask to check if it's a file
326  isdir=false; //it's a file
327  } else {
328  cerr<<"Check input path: stat says LAPPD path is neither file nor directory..?"<<endl;
329  return nullptr;
330  }
331  } else {
332  // errors could be because this is a file pattern: e.g. wcsim_0.4*.root Treat as a file.
333  isdir=false;
334  }
335 
336  TChain * t = new TChain("LAPPDTree","");
337  // if it's a directory
338  if(isdir){
339  const char* ext = ".root";
340  TString nextfilepattern;

342  TList *subfolders = dir.GetListOfFiles(); // despite name returns files and folders
343 
344  // Add subfolders if requested
345  if(addsubfolders&&subfolders&&(subfolders->GetEntries()>2)){ // always has '.' and '..'
346  if(verbose>2) cout<<"looping over subfolders"<<endl;
347  TSystemDirectory *subfolder;

359  } // end if subfolders are to be added

363  nextfilepattern = std::string(inputdir) + std::string("/wcsim_lappd_*.root"); // XXX
