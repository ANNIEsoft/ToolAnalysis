83  RealDetectors.resize(RealDetectors.size()+1);

85  DetectorElements.emplace(thedetel,RealDetectors.size()-1); // maps det. element to vector index
86  detectorsetindex = RealDetectors.size()-1;

94  return false;
95  } else {
96  Detectors.at(detectorsetindex)->emplace(detin.GetDetectorID(), detin);
97  }
98  // increment counters of the number of detectors in each set
99  if(detectorcounts.count(thedetel)==0){
100  detectorcounts.emplace(thedetel,1);
