00107     bool AddDetector(Detector detin){
00108         // quickly check if we've allocated this DetectorKey
00109         if(DetectorKeys.count(detin.GetDetectorID())){
00110             std::cerr<<"Geometry error! AddDetector called with non-unique DetectorKey "
00111                      <<detin.GetDetectorID()<<std::endl;
00112             return false;
00113         } else {
00114             DetectorKeys.emplace(detin.GetDetectorID(),1);
00115         }
00116         
00117         // Pass a pointer to it's owning geometry to this Detector
00118         // we need to do this before calling `emplace` as that must do a copy-construction
00119         detin.SetGeometryPtr(this);
00120         
00121         std::string thedetel = detin.GetDetectorElement();
00122         if(Detectors.count(thedetel)==0){
00123             // this is a new detector element - create a new entry in the Detectors map
00124             RealDetectors.emplace(thedetel,std::map<unsigned long,Detector>{});
00125             Detectors.emplace(thedetel,std::map<unsigned long,Detector*>{});
00126         }
00127         RealDetectors.at(thedetel).emplace(detin.GetDetectorID(), detin);
00128         Detectors.at(thedetel).emplace(detin.GetDetectorID(),
00129                                         &RealDetectors.at(thedetel).at(detin.GetDetectorID()));
00130         
00131         return true;
