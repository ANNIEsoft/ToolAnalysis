00176 static double compute_variance(const double mean, const std::vector<double>& numbers){
00177 
00178   auto add_square = [mean](double sum, int i)
00179     {
00180       auto d = i - mean;
00181       return sum + d*d;
00182     };
00183   double total = std::accumulate(numbers.begin(), numbers.end(), 0.0, add_square);
00184   if (numbers.size() == 1) {
00185     //std::cout <<"compute variance: return 0"<<std::endl;
00186     return 0.;
00187   }
00188   //std::cout <<"compute variance: return "<<sqrt(total / (numbers.size() - 1))<<std::endl;
00189   return sqrt(total/ (numbers.size() - 1));
