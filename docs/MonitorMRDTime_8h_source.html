171 static double compute_variance(const double mean, const std::vector<double>& numbers){
172 
173 /*
174  if (numbers.size() <= 1u)
175  std::cout <<"compute variance: returning not a number..."<<std::endl;
176  return std::numeric_limits<double>::quiet_NaN();
177  */
178 
179  auto add_square = [mean](double sum, int i)
180  {
181  auto d = i - mean;
182  return sum + d*d;
183  };
184  double total = std::accumulate(numbers.begin(), numbers.end(), 0.0, add_square);
185  if (numbers.size() == 1) {
186  //std::cout <<"compute variance: return 0"<<std::endl;
187  return 0.;
188  }
189  //std::cout <<"compute variance: return "<<sqrt(total / (numbers.size() - 1))<<std::endl;
190  return sqrt(total/ (numbers.size() - 1));
191 }
